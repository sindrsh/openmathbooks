\vedlegg{Functions} \label{FuncAttach}
\reg[Domain]{
	\outl{The domain} of a function \( f(x) \) is the \( x \)-values for which \( f(x) \) is valid.
}\regv

\reg[Range]{
	\outl{The range} of a function \( f(x) \) is all the values \( f(x) \) can take. The range is determined by the function's expression and its domain.
}
\regv

\reg[Proportional Quantities]{
	Given a constant \( a \) and two variables \( x \) and \( y \). If 
	\[ a = \frac{x}{y} \]
	then \( x \) and \( y \) are \outl{proportional} quantities.
}\regv

\reg[Inversely Proportional Quantities]{
	Given a constant \( a \) and two variables \( x \) and \( y \). If 
	\[ a=xy \]
	then \( x \) and \( y \) are \outl{inversely proportional} quantities.
}\regv

\reg[Polynomial Functions]{
	A polynomial function is a function consisting of a sum of powers with positive exponents and a variable as a base. \vsk
	
	Polynomial functions have sub-titles determined by the highest exponent in the function expression. For constants \( a \), \( b \), \( c \), and \( d \), and a variable \( x \), we have
	\begin{center}
		\begin{tabular}{|r|l|} \hline
			\textbf{function expression} & \textbf{function name} \\ \hline
			\( ax+b \) & 1st degree function (linear) \\
			\( ax^2+bx+c \) & 2nd degree function (quadratic) \\
			\( ax^3+bx^2+cx+d \) & 3rd degree function (cubic) \\ \hline
		\end{tabular}
	\end{center}
}
\spr{
	\outl{The validity domain} of a function states the \( x \)-values for which the function makes sense in the practical situation it's used in.
}
